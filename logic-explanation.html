// ログイン処理の疑似コード

FUNCTION login(email, password)

  // 入力値チェック
  IF email が空 THEN
    エラー「メールアドレスを入力してください」を追加
  END IF

  IF password が空 THEN
    エラー「パスワードを入力してください」を追加
  END IF

  IF エラーリストが空でない THEN
    RETURN ログイン画面（エラー表示）
  END IF

  // ユーザー検索
  user = userMapper.selectByEmail(email)

  IF user が null THEN
    エラー「メールアドレスまたはパスワードが正しくありません」を追加
    RETURN ログイン画面（エラー表示）
  END IF

  // パスワード照合
  IF BCrypt.matches(password, user.password) が false THEN
    エラー「メールアドレスまたはパスワードが正しくありません」を追加
    RETURN ログイン画面（エラー表示）
  END IF

  // 認証成功
  SecurityContext に user 情報を保存
  RETURN リダイレクト("/tasks")

END FUNCTION

// タスク登録のバリデーション処理

FUNCTION createTask(taskForm, bindingResult)

  // Bean Validationのエラーチェック
  IF bindingResult.hasErrors() THEN
    RETURN 入力画面（フォーム + エラーメッセージ）
  END IF

  // 業務ロジックバリデーション

  // 期限の妥当性チェック
  IF taskForm.dueDate が過去の日付 THEN
    bindingResult.rejectValue("dueDate", "error.dueDate", "期限は今日以降の日付を指定してください")
  END IF

  // タイトルの重複チェック（同一ユーザー内）
  existingTask = taskMapper.selectByTitleAndUserId(taskForm.title, userId)
  IF existingTask が存在する THEN
    bindingResult.rejectValue("title", "error.title", "同じタイトルのタスクが既に存在します")
  END IF

  // エラーがあれば入力画面に戻る
  IF bindingResult.hasErrors() THEN
    RETURN 入力画面（フォーム + エラーメッセージ）
  END IF

  // 登録処理
  task = FormからEntityに変換(taskForm)
  task.userId = ログインユーザーID
  task.createdAt = 現在日時
  taskMapper.insert(task)

  RETURN リダイレクト("/tasks")（成功メッセージ付き）

END FUNCTION

【MyBatis動的SQLの考え方】
・keyword がある場合のみ title / content の LIKE 条件を追加
・categoryId がある場合のみ category_id 条件を追加
・status がある場合のみ status 条件を追加
・startDate がある場合のみ 開始日条件を追加
・endDate がある場合のみ 終了日条件を追加
・各条件は <if test="..."> で必要なときだけ WHERE句に連結する

複合検索のロジック 必須
疑似コードでの検索条件組立
// 複合検索のロジック（MyBatis XML形式）

<select id="search" resultType="Task">
  SELECT * FROM tasks
  WHERE user_id = #{userId}

  <!-- キーワード検索 -->
  <if test="keyword != null and keyword != ''">
    AND (title LIKE CONCAT('%', #{keyword}, '%')
    OR content LIKE CONCAT('%', #{keyword}, '%'))
  </if>

  <!-- カテゴリ絞り込み -->
  <if test="categoryId != null">
    AND category_id = #{categoryId}
  </if>

  <!-- ステータス絞り込み -->
  <if test="status != null and status != ''">
    AND status = #{status}
  </if>

  <!-- 期間絞り込み（開始日） -->
  <if test="startDate != null">
    AND due_date >= #{startDate}
  </if>

  <!-- 期間絞り込み（終了日） -->
  <if test="endDate != null">
    AND due_date <= #{endDate}
  </if>

  <!-- ソート条件 -->
  ORDER BY
  <choose>
    <when test="sortBy == 'dueDate'">due_date ASC</when>
    <when test="sortBy == 'priority'">priority DESC</when>
    <otherwise>created_at DESC</otherwise>
  </choose>
</select>

FUNCTION updateTask(taskId, taskForm, loginUser)

  // タスクを取得
  task = taskMapper.selectById(taskId)

  // 存在チェック
  IF task が null THEN
    THROW NotFoundException("タスクが見つかりません")
  END IF

  // 権限チェック
  IF loginUser.role が "ADMIN" でない
    AND task.userId ≠ loginUser.id THEN
    THROW AccessDeniedException("このタスクを編集する権限がありません")
  END IF

  // 更新処理（権限OK）
  task = FormからEntityに変換(taskForm)
  task.id = taskId
  task.userId = 元のtask.userId
  task.createdAt = 元のtask.createdAt
  task.updatedAt = 現在日時
  taskMapper.update(task)

  RETURN 成功

END FUNCTION

【遷移可否一覧】
TODO → DOING　　　 ：○　作業を開始した場合
TODO → HOLD　　　　：×　未着手から直接保留にはしない想定
TODO → DONE　　　　：×　未着手から直接完了にはしない想定
TODO → CANCELLED　 ：○　着手前に中止判断した場合

DOING → TODO　　　 ：○　作業開始を取り消して未着手へ戻す場合
DOING → HOLD　　　 ：○　一時的に作業停止する場合
DOING → DONE　　　 ：○　作業完了時
DOING → CANCELLED　 ：○　途中で中止判断した場合

HOLD → TODO　　　　：○　保留解除後、未着手として再管理する場合
HOLD → DOING　　　 ：○　作業再開時
HOLD → DONE　　　　：×　保留から直接完了にはしない想定
HOLD → CANCELLED　 ：○　保留中に中止判断した場合

DONE → TODO　　　　：×　原則不可
DONE → DOING　　　 ：○　完了後に再対応が必要になった場合
DONE → HOLD　　　　：×　原則不可
DONE → CANCELLED　 ：×　完了後の中止は不可

CANCELLED → TODO　 ：○　中止取消で再登録する場合
CANCELLED → DOING ：×　原則不可
CANCELLED → HOLD　 ：×　原則不可
CANCELLED → DONE　 ：×　原則不可

【遷移条件】
・TODO → DOING：担当者が作業を開始したとき
・DOING → HOLD：作業継続が一時的にできないとき
・HOLD → DOING：保留理由が解消し、再開可能になったとき
・DOING → DONE：必要な作業が完了したとき
・TODO / DOING / HOLD → CANCELLED：不要化、依頼取消、対応中止の判断があったとき
・DONE → DOING：完了後に修正や追加対応が必要になったとき
・CANCELLED → TODO：中止判断を取り消して再度管理対象に戻すとき

遷移可否の判定ロジック 必須
状態遷移が許可されるかの判定処理
// ステータス遷移の判定ロジック（Enum想定）

FUNCTION canTransition(currentStatus, newStatus)

  // 遷移ルールの定義
  allowedTransitions = {
    TaskStatus.TODO → [TaskStatus.DOING, TaskStatus.CANCELLED],
    TaskStatus.DOING → [TaskStatus.TODO, TaskStatus.HOLD, TaskStatus.DONE, TaskStatus.CANCELLED],
    TaskStatus.HOLD → [TaskStatus.DOING, TaskStatus.CANCELLED],
    TaskStatus.DONE → [], // 遷移不可
    TaskStatus.CANCELLED → [] // 遷移不可
  }

  // 現在のステータスから遷移可能なステータスを取得
  allowed = allowedTransitions[currentStatus]

  // 遷移可否を判定
  IF newStatus が allowed に含まれる THEN
    RETURN true
  ELSE
    RETURN false
  END IF

END FUNCTION

// ステータス変更処理

FUNCTION changeStatus(taskId, newStatus)

  task = taskMapper.selectById(taskId)

  IF task が null THEN
    THROW NotFoundException("タスクが見つかりません")
  END IF

  IF NOT canTransition(task.status, newStatus) THEN
    THROW IllegalStateException(
      "「" + task.status + "」から「" + newStatus + "」への遷移はできません"
    )
  END IF

  task.status = newStatus
  task.updatedAt = 現在日時

  // 完了時は完了日時を記録
  IF newStatus == TaskStatus.DONE THEN
    task.completedAt = 現在日時
  ELSE
    // 完了以外へ変更された場合は必要に応じてクリア
    task.completedAt = null
  END IF

  taskMapper.updateStatus(task)

END FUNCTION
